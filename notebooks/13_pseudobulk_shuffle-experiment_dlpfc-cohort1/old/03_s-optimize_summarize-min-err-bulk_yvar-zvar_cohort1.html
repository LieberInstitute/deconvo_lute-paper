Get minima by label

Gets one unique minimum neuron error per data label.

length(unique(df.res$sample.label))
## [1] 65
labels.vector <- unique(df.res$sample.label)
df.min <- do.call(rbind, lapply(labels.vector, function(label.iter){
  df.iter <- df.res[df.res$sample.label==label.iter,]
  df.iter[df.iter$error.neuron == min(df.iter$error.neuron)[1],,drop=F]
}))
df.min$min.error.neuron <- df.min$error.neuron
# filter outlying high error
dim(df.min)
## [1] 3672   23
df.min <- df.min[df.min$min.error.neuron<0.1,]
dim(df.min)
## [1] 72 23
